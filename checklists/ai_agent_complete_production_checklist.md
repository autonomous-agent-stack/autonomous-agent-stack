## 🚀 快速检查脚本

```bash
#!/bin/bash
# production_check.sh

echo "=== AI Agent 生产检查 ==="

# 1. 代码质量
echo "1. 代码质量..."
pytest --cov=app --cov-report=term-missing
coverage=$(coverage report | grep TOTAL | awk '{print $4}' | sed 's/%//')
if [ $coverage -lt 80 ]; then
  echo "❌ 测试覆盖率不足: $coverage%"
else
  echo "✅ 测试覆盖率: $coverage%"
fi

# 2. 安全检查
echo "2. 安全检查..."
bandit -r app/ -f json > security_report.json
issues=$(cat security_report.json | jq '.results | length')
if [ $issues -gt 0 ]; then
  echo "❌ 发现安全问题: $issues 个"
else
  echo "✅ 安全检查通过"
fi

# 3. 性能测试
echo "3. 性能测试..."
locust -f locustfile.py --headless -u 100 -r 10 -t 60s --host http://localhost:8000

# 4. Docker 构建
echo "4. Docker 构建..."
docker build -t ai-agent:test .
size=$(docker images ai-agent:test --format "{{.Size}}")
echo "镜像大小: $size"

# 5. 健康检查
echo "5. 健康检查..."
curl -f http://localhost:8000/health || echo "❌ 健康检查失败"

echo "=== 检查完成 ==="
```

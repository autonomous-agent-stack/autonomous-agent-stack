# CrewAI 多 Agent 示例
from crewai import Agent, Task, Crew

researcher = Agent(
    role="研究员",
    goal="搜索和分析信息",
    backstory="专业的 AI 研究员",
    tools=[search_tool, scrape_tool]
)

writer = Agent(
    role="作者",
    goal="撰写高质量文章",
    backstory="经验丰富的技术作者",
    tools=[write_tool]
)

crew = Crew(
    agents=[researcher, writer],
    tasks=[research_task, write_task]
)

result = crew.kickoff()
